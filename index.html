} function doPost(e) { var output = { status: 'error', message: 'Unknown request' }; try { var params = JSON.parse(e.postData.contents); var action = params.action; // ตรวจสอบและสร้างฐานข้อมูลตารางอัตโนมัติทุกครั้งที่มี Request checkAndInitSheets_(); } else if (action === 'recordAttendance') { output = recordAttendance_(params); } else if (action === 'getDashboardData') { output = getDashboardData_(); } // --- Admin CRUD Actions --- else if (action === 'verifyAdmin') { if (params.password === ADMIN_PASSWORD) { output = { status: 'success' }; } else { output = { status: 'error', message: 'รหัสผ่านผู้ดูแลระบบไม่ถูกต้อง' }; } } else if (action === 'getEmployees') { output = { status: 'success', data: getEmployeesList_() }; } else if (action === 'saveEmployee') { output = saveEmployee_(params.data); } else if (action === 'deleteEmployee') { output = deleteEmployee_(params.empId); } else if (action === 'getLocations') { output = { status: 'success', data: getLocationsList_() }; } else if (action === 'saveLocation') { output = saveLocation_(params.data); } else if (action === 'deleteLocation') { output = deleteLocation_(params.locId); } } catch (error) { output = { status: 'error', message: error.message || error.toString() }; } return ContentService.createTextOutput(JSON.stringify(output)) .setMimeType(ContentService.MimeType.JSON); } // 3. ตารางจุดลงเวลา/สาขา (Locations) var locSheet = ss.getSheetByName(SHEET_NAME_LOCATIONS); if (!locSheet) { locSheet = ss.insertSheet(SHEET_NAME_LOCATIONS); locSheet.appendRow(['รหัสสาขา', 'ชื่อสาขา/สถานที่', 'ละติจูด', 'ลองจิจูด', 'รัศมีอนุญาต (เมตร)']); locSheet.getRange(1, 1, 1, 5).setFontWeight('bold').setBackground('#c62828').setFontColor('#ffffff'); locSheet.setFrozenRows(1); } } function verifyEmployee_(empId, pin) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_EMPLOYEES); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0].toString().trim() === empId.toString().trim()) { if (data[i][3].toString().trim() === pin.toString().trim()) { if (data[i][4].toString().trim() !== 'Active') { return { status: 'error', message: 'รหัสพนักงานนี้ถูกระงับการใช้งาน' }; } // ตรวจสอบประวัติการลงเวลาของวันนี้ เพื่อเช็คสถานะล่าสุดและป้องกันการซ้ำซ้อน var lastStatus = getTodayLastStatus_(empId); return { status: 'success', empName: data[i][1], department: data[i][2], lastStatus: lastStatus }; } else { return { status: 'error', message: 'รหัส PIN 4 หลักไม่ถูกต้อง' }; } } } return { status: 'error', message: 'ไม่พบข้อมูลรหัสพนักงานนี้ในระบบ' }; } function getTodayLastStatus_(empId) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 'เข้างาน' หรือ 'ออกงาน' break; } } } return lastStatus; } function saveSelfieToDrive_(base64Data, empId) { try { var folderName = 'Attendance_Selfies'; var folders = DriveApp.getFoldersByName(folderName); // ทำการรีเช็ค PIN บนเซิร์ฟเวอร์อีกรอบเพื่อความปลอดภัยสูงสุุด var verification = verifyEmployee_(empId, pin); if (verification.status === 'error') { throw new Error(verification.message); } var userLat = parseFloat(params.userLat); } function getEmployeesList_() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_EMPLOYEES); var data = sheet.getDataRange().getValues(); var list = []; for (var i = 1; i < data.length; i++) { list.push({ empId: data[i][0], name: data[i][1], department: data[i][2], pin: data[i][3], status: data[i][4] }); } return list; } function saveEmployee_(emp) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_EMPLOYEES); var data = sheet.getDataRange().getValues(); var empId = emp.empId.toString().trim(); // สุ่ม PIN 4 หลักอัตโนมัติหากไม่มีการระบุมา var pin = emp.pin ? emp.pin.toString().trim() : Math.floor(1000 + Math.random() * 9000).toString(); } else { // ตรวจสอบไอดีซ้ำก่อนจะบันทึกรายใหม่ for (var i = 1; i < data.length; i++) { if (data[i][0].toString().trim() === empId) { return { status: 'error', message: 'รหัสพนักงานนี้มีในระบบแล้ว' }; } } // เพิ่มพนักงานรายใหม่ sheet.appendRow([empId, emp.name, emp.department, pin, 'Active']); return { status: 'success', message: 'เพิ่มพนักงานสำเร็จ! รหัส PIN สุ่มอัตโนมัติของคุณคือ: ' + pin }; } } function deleteEmployee_(empId) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_EMPLOYEES); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0].toString().trim() === empId.toString().trim()) { sheet.deleteRow(i + 1); return { status: 'success', message: 'ลบข้อมูลพนักงานเรียบร้อย' }; } } return { status: 'error', message: 'ไม่พบพนักงานที่ต้องการลบ' }; } function getLocationsList_() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_LOCATIONS); } return list; } function saveLocation_(loc) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_LOCATIONS); var data = sheet.getDataRange().getValues(); var locId = loc.locId ? loc.locId.toString().trim() : 'LOC' + new Date().getTime(); return { status: 'success', message: 'บันทึกจุดพิกัดลงเวลาใหม่เรียบร้อย' }; } } function deleteLocation_(locId) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME_LOCATIONS); } } return { status: 'error', message: 'ไม่พบข้อมูลสาขาที่เลือก' }; } function getDashboardData_() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var attSheet = ss.getSheetByName(SHEET_NAME_ATTENDANCE); todayLogs.push({ time: logTime, empId: attData[i][1], name: attData[i][2], department: attData[i][3], type: type, location: attData[i][5], distance: attData[i][6], photo: attData[i][9] }); } } } return { status: 'success', totalEmployees: empCount, todayActiveCount: uniqueStaffToday.size, todayIn: inCount, todayOut: outCount, logs: todayLogs }; } /* ---------- คำนวณระยะทางสูตรเกรทเซอร์เคิล (เมตร) ---------- */ function distanceMeters_(lat1, lng1, lat2, lng2) { var R = 6371000; var toRad = function (d) { return d * Math.PI / 180; }; var dLat = toRad(lat2 - lat1); var dLng = toRad(lng2 - lng1); var a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLng / 2) * Math.sin(dLng / 2); return R * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)));
